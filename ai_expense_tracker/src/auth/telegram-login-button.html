Telegram Login Example

This is an example of how to integrate the Telegram Login widget into your website.

Replace YOUR_BOT_NAME with your actual Telegram bot's username (without the @ symbol).

Replace YOUR_AUTH_URL with your backend validation endpoint, e.g., "https://your-api.com/api/auth/telegram/validate".

Integration Code

<script async src="https://telegram.org/js/telegram-widget.js?22" 
        data-telegram-login="YOUR_BOT_NAME" 
        data-size="large" 
        data-auth-url="YOUR_AUTH_URL" 
        data-request-access="write"></script>

Optional Parameters

<script async src="https://telegram.org/js/telegram-widget.js?22" 
        data-telegram-login="YOUR_BOT_NAME" 
        data-size="large"               // large, medium, small
        data-auth-url="YOUR_AUTH_URL"   // Your backend validation endpoint
        data-request-access="write"     // write or read
        data-userpic="false"            // Show user picture (true or false)
        data-radius="10"                // Corner radius in pixels
        data-onauth="onTelegramAuth(user)"  // JS callback function
        data-corner-radius="20"         // Alternative corner radius parameter
        data-button-radius="10"></script>

JavaScript Callback Example

<script type="text/javascript">
  function onTelegramAuth(user) {
    // Send the received data to your backend for validation
    fetch('YOUR_AUTH_URL', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(user)
    })
    .then(response => response.json())
    .then(data => {
      // Store the received token
      localStorage.setItem('auth_token', data.accessToken);
      // Redirect or update UI
      window.location.href = '/dashboard';
    })
    .catch(error => {
      console.error('Authentication Error:', error);
    });
  }
</script>
